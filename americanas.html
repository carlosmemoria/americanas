Gerando um gráfico de pizza e de barras com todas as entradas do dataset¶

#Distribuio dos dados da classe Y (No existe ocorrncia = n, Existe ocorrncia = e)

n = pd.value_counts(df['target']) [0]
e = pd.value_counts(df['target']) [1]

condTrue = 'No h ocorrncia do evento que desejamos prever em '
condFalse = 'Existe ocorrncia do evento que desejamos prever '
resposta = 'linhas'

tam = len(df)

pie = pd.DataFrame([['H ocorrncia',n],['No h ocorrncia',e]],columns=['Target' , 'Quant'])

def pie_chart(data,col1,col2,title): 
    labels = {'No':0,'Sim':1}
    sizes = data[col2]
    colors = ['#e5ffcc', '#f80032']

    plt.pie(sizes, labels=labels, colors=colors,
                autopct='%1.1f%%', shadow=True, startangle=140, labeldistance =1.2)
    plt.title( title )
    
    plt.axis('equal')
    plt.show()

pie_chart(pie,'Target' , 'Quant','Distribuio Percentual quanto a existncia ou no de ocorrncia')

print("\n")
print("{}{}{}".format(condTrue,n," "+ resposta))
print("{}{}{}".format(condFalse,e," "+ resposta))
print("\n")

plt.bar(pie.Target,pie.Quant, color = ['#e5ffcc', '#f80032'])
plt.title("Distribuio quanto a existncia ou no de ocorrncia")
plt.xlabel("Existe ocorrncia?")
plt.ylabel('Quantidade de Registros')
plt.show()
